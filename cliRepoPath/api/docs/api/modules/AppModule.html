import { MiddlewareConsumer, Module, OnModuleDestroy, OnModuleInit } from '@nestjs/common';

import { PagesModule } from './modules/pages/pages.module';

//For loggin memory used in endpoint
import { APP_INTERCEPTOR } from '@nestjs/core';
import { MemoryInterceptor } from '@app/common/interceptors/memory.record.interceptor';

@Module({
  imports: [
    ServeStaticModule.forRoot({
      rootPath: join(process.cwd(), 'docs/api'), // Ensure this points to the correct folder
      serveRoot: '/api-docs', // custom path
    }),
    ServeStaticModule.forRoot({
      rootPath: join(process.cwd(), 'docs/cms-api'), // Ensure this points to the correct folder
      serveRoot: '/cms-api-docs', // custom path
    }),
    ServeStaticModule.forRoot({
      rootPath: join(process.cwd(), 'docs/combined'), // Ensure this points to the correct folder
      serveRoot: '/docs', // custom path
    }),
    MulterModule.registerAsync({
      inject: [ConfigService],
      useFactory: (configService: ConfigService) => ({
        dest: configService.get<string>('MULTER_UPLOAD_DEST'), // Default to './uploads' if not set in config
      }),
    }),
    BullModule.forRootAsync({
      useFactory: async (configService: ConfigService) => {
        return {
          connection: {
            host: configService.get<string>('REDIS_HOST'),
            port: configService.get<number>('REDIS_PORT'),
            password: configService.get<string>('REDIS_PASSWORD'),
          },
        };
      },
      inject: [ConfigService],
    }),
    BullBoardModule.forRootAsync({
      inject: [ConfigService],
      useFactory: (configService: ConfigService) => ({
        route: `${configService.get<string>('BULL_BOARD_PATH')}`,
        adapter: ExpressAdapter, // Or FastifyAdapter from `@bull-board/fastify`
      }),
    }),

    /** providers */
    ConfigProvider,

    //DatabaseProvider,
    CronProvider,

    /** modules */
    AuthModule,
    StripePaymentModule,
    UsersModule,
    PagesModule,
    StripeSubscriptionModule,
    SubscriptionModule,
    FakeApiModule,
    FeedsModule,
    FeedCommentsModule,
    FeedRepliesModule,
    CoinManagementModule,
    NotificationsModule,
    CronModule,
    ChimeModule,
    AgoraModule,
    CometChatModule,
    FaqModule,
    ContactsModule,
    SiteSettingsModule,
    //SearchModule,
    RapidIdModule,
    LoggerModule,
    StripeConnectModule,
    PermissionModule,
    I18nModule.forRoot({
      fallbackLanguage: 'en',
      loaderOptions: {
        path: path.join(__dirname, '/i18n/'),
        watch: true,
      },
      resolvers: [
        GraphQLWebsocketResolver,
        { use: QueryResolver, options: ['lang'] },
        new HeaderResolver(['x-lang']),
        AcceptLanguageResolver,
      ],
    }),

    GraphQLModule.forRoot<ApolloDriverConfig>({
      validationRules: [depthLimit(5)], // Limit query depth to 5 levels
      driver: ApolloDriver,
      autoSchemaFile: true,
      resolvers: { JSON: GraphQLJSON },
      sortSchema: true,
      path: '/api',
      playground: false,
      introspection: true,
      context: ({ req }) => ({ req }),
      csrfPrevention: true,
      plugins: [ApolloServerPluginLandingPageLocalDefault()],
      include: [
        AuthModule,
        StripePaymentModule,
        UsersModule,
        PagesModule,
        StripeSubscriptionModule,
        SubscriptionModule,
        FakeApiModule,
        FeedsModule,
        FeedCommentsModule,
        FeedRepliesModule,
        CoinManagementModule,
        NotificationsModule,
        ChimeModule,
        AgoraModule,
        CometChatModule,
        FaqModule,
        ContactsModule,
        SiteSettingsModule,
        RapidIdModule,
        StripeConnectModule,
      ],

      formatError: (error: any) => {
        if (error instanceof ApolloError) {
          return error;
        }
        return formatGraphQLError(error);
      },
    }),
    EmailModule.forRoot(),
    MongooseModule.forRootAsync({
      useFactory: async (configService: ConfigService) => ({
        uri: encodeURI(configService.get('DB_CONNECTION_STRING')),
      }),
      inject: [ConfigService],
    }),
  ],

  //For loggin memory used in endpoint
  providers: [
    {
      provide: APP_INTERCEPTOR,
      useClass: MemoryInterceptor,
    },
  ],
})
export class AppModule {
  configure(consumer: MiddlewareConsumer) {
    consumer.apply(RequestLoggerMiddleware).forRoutes('*'); // Apply to all routes, or specify particular routes
  }
}
