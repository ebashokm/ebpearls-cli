@Module({
  imports: [
    // Configuration Module
    ConfigModule.forRoot({
      isGlobal: true,
      envFilePath: './apps/cms-api/.env',
      validationSchema: configValidationSchema,
      //expandVariables: true,
      //validationSchema: configValidationSchema,
    }),

    // I18n Module
    I18nModule.forRoot({
      fallbackLanguage: 'en',
      loaderOptions: {
        path: path.join(__dirname, '/i18n/'),
        watch: true,
      },
      resolvers: [
        GraphQLWebsocketResolver,
        { use: QueryResolver, options: ['lang'] },
        AcceptLanguageResolver,
      ],
    }),
    // GraphQL Module
    GraphQLModule.forRoot<ApolloDriverConfig>({
      validationRules: [depthLimit(5)], // Limit query depth to 5 levels
      driver: ApolloDriver,
      autoSchemaFile: true,
      path: '/cms-api',
      playground: false,
      plugins: [ApolloServerPluginLandingPageLocalDefault()],
      formatError: formatGraphQLError,
    }),
    EventEmitterModule.forRoot(), //for event emitter and listener

    // Custom Application Modules
    DataAccessModule,
    AdminModule,
    AuthModule,
    AdvancePageModule,
    AppUserModule,
    FaqModule,
    EmailTemplateModule,
    HomePageTemplateModule,
    TestimonialsModule,
    MenuModule,
    TaxonomyModule,
    SubscriptionProductsModule,
    SettingModule,
    BusinessModule,
    PageModule,
    PageWithVersionModule,
    SiteSettingsModule,
    LoggerModule,
    RoleModule,
    PermissionModule,
    // Mongoose Module
    // MongooseModule.forRootAsync({
    //   useFactory: async (configService: ConfigService) => ({
    //     uri: process.env.DATABASE_URI,
    //     useNewUrlParser: true,
    //     useUnifiedTopology: true,
    //   }),
    //   inject: [ConfigService],
    // }),
    MongooseModule.forRootAsync({
      useFactory: async (configService: ConfigService) => ({
        uri: encodeURI(configService.get('DB_CONNECTION_STRING')),
      }),
      inject: [ConfigService],
    }),
    EmailModule.forRoot(),
  ],
  providers: [
    {
      provide: APP_INTERCEPTOR,
      useClass: MemoryInterceptor,
    },
  ],
})
export class AppModule {
  configure(consumer: MiddlewareConsumer) {
    consumer.apply(RequestLoggerMiddleware).forRoutes('*'); // Apply to all routes, or specify particular routes
  }
}
