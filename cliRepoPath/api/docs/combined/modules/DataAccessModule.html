import { Module } from '@nestjs/common';
import { DataAccessService } from './data-access.service';
import { MongooseModule } from '@nestjs/mongoose';
import { Contacts, ContactsRepository, contactsSchema } from './contacts';
import { User, UserSchema, UsersRepository } from './user';

@Module({
  imports: [
    // Register schemas with Mongoose
    MongooseModule.forFeature([
      { name: Contacts.name, schema: contactsSchema },
      { name: User.name, schema: UserSchema },
      // Add other schemas here
    ]),
  ],
  providers: [UsersRepository, ContactsRepository, DataAccessService],
  exports: [
    UsersRepository,
    ContactsRepository,
    MongooseModule,
    DataAccessService,
  ],
})
export class DataAccessModule {}
