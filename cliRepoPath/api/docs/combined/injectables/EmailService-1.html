  async sendForgotPasswordEmail(to: string, values) {
    console.log(to);

    const templateData = await this.emailTemplateRepository.findOne({
      slug: 'forgot-password',
    });

    const content = replacePlaceholders(templateData.body, values);
    //const content = parseContent({ content: templateData.body, ...values });

    const htmlContent = generateEmailTemplate(content);

    await this.sesService.sendEmail(to, templateData.subject, htmlContent);
    return true;
