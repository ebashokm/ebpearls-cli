import { BadRequestException, HttpException, HttpStatus, Injectable } from '@nestjs/common';
import { Args } from '@nestjs/graphql';
import { UserSubscriptionRepository } from '@app/data-access';
import { UsersRepository } from '@app/data-access';
import { AndroidReceipt, IosReceipt } from './dto/input/validate-receipt.input';
import { PurchasePlatform } from './enum/subscription-platform.enum';

import { InappSubscriptionService } from '@app/inapp-subscription';
// import { InappSubscriptionService } from 'subscription-package-node'; // use as a package
import { SubscriptionLogRepository } from '@app/data-access';
import { IAP_SUBSCRIPTION, PRO_SUBSCRIPTION_ID } from '../../constants';
import { SubscriptionFrequency } from './enum/subscription-frequency.enum';
import { SubscriptionStatus } from './enum/subscription-status.enum';
import { I18nService } from 'nestjs-i18n';
import { ENVIRONMENT, NOTIFICATION_TYPE } from './constants';

@Injectable()
export class SubscriptionService {
  constructor(
    private readonly iap: InappSubscriptionService,

    private readonly userRepository: UsersRepository,
    private readonly subscriptionLogRepo: SubscriptionLogRepository,
    private readonly i18nService: I18nService,
  ) {}

  /**
   * ${1:Description placeholder}
   *
   * @async
   * @param {*} user
   * @param {IosReceipt} receiptData
   * @returns {Promise<boolean>}
   */
  async verifyIosSubscription(user, receiptData: IosReceipt) {
    try {
      const data = await this.iap.validateIosReceipt(receiptData.receipt);

      if (data instanceof Error) {
        throw new HttpException(
          data?.message || this.i18nService.t('subscription.receipt_error'),
          HttpStatus.UNPROCESSABLE_ENTITY,
        );
      }

      const purchasedData = data?.purchaseData[0];
      const validatedData = data?.validatedData;

      // validationResponse contains sandbox: true/false for Apple and Amazon
      // Android we don't know if it was a sandbox account
      const environment = validatedData.sandbox ? ENVIRONMENT.SANDBOX : ENVIRONMENT.PRODUCTION;

      const subscriptionData = {
        purchasePlatform: PurchasePlatform.IOS,
        istrailPeriod: !!purchasedData?.isTrial,
        autoRenewStatus: !!purchasedData?.autoRenewing,
        originalTransactionId: purchasedData?.originalTransactionId,
        environment,
        purchaseDate: purchasedData?.originalPurchaseDateMs
          ? new Date(purchasedData?.originalPurchaseDateMs)
          : null,
        cancellationDate: purchasedData?.cancellationDate
          ? new Date(purchasedData?.cancellationDate)
          : null,
        expirationDate: purchasedData?.expiresDateMs
          ? new Date(purchasedData?.expiresDateMs)
          : null,
      };

      const subscription = await this.subscriptionRepository.findOne({
        productId: purchasedData?.productId,
        originalTransactionId: purchasedData?.originalTransactionId,
        expirationDate: { $gte: new Date() },
      });

      if (subscription) {
        // diff user
        if (subscription.userId !== user.userId.toString()) {
          throw new BadRequestException(
            this.i18nService.t('subscription.package_already_subscribed_from_another_user'),
          );
        }
        // same user
        await this.subscriptionRepository.updateOne(
          {
            userId: user.userId.toString(),
            productId: purchasedData?.productId,
            originalTransactionId: purchasedData?.originalTransactionId,
          },
          {
            ...subscriptionData,
            receipt: receiptData.receipt,
            productId: purchasedData?.productId,
            transactionId: purchasedData?.transactionId,
          },
        );
      } else {
        // check if another user has already subscribed from this account
        const deviceSubscription = await this.subscriptionRepository.findOne({
          originalTransactionId: purchasedData?.originalTransactionId,
          userId: { $ne: user.userId.toString() },
        });

        if (deviceSubscription) {
          // not expired
          if (new Date(deviceSubscription.expirationDate) > new Date()) {
            throw new BadRequestException(
              this.i18nService.t('subscription.package_already_subscribed_from_this_account'),
            );
          } else {
            // expired
            await this.subscriptionRepository.updateOne(
              {
                originalTransactionId: purchasedData?.originalTransactionId,
              },
              {
                ...subscriptionData,
                userId: user.userId.toString(),
                receipt: receiptData.receipt,
                productId: purchasedData?.productId,
                transactionId: purchasedData?.transactionId,
              },
            );

            return true;
          }
        } else {
          // same user case
          // check if current user has subscribed the same package from another account
          const userSubscription = await this.subscriptionRepository.findOne({
            userId: user.userId.toString(),
            productId: purchasedData?.productId,
            expirationDate: { $gte: new Date() },
          });
          if (userSubscription) {
            throw new BadRequestException(
              this.i18nService.t('subscription.package_already_subscribed_from_another_account'),
            );
          }
          await this.subscriptionRepository.create({
            userId: user.userId.toString(),
            receipt: receiptData.receipt,
            productId: purchasedData?.productId,
            transactionId: purchasedData?.transactionId,
            ...subscriptionData,
          });
        }
      }

      let priceAmount; // in cents
      for (const sub of IAP_SUBSCRIPTION) {
        if (sub.productId === purchasedData?.productId) {
          priceAmount = sub.price.amount * 100;
        }
      }

      const membershipAppliedAt = purchasedData?.purchaseDateMs
        ? new Date(purchasedData?.purchaseDateMs)
        : new Date();

      const logData = {
        userId: user.userId.toString(),
        username: user.authProviderId || '',
        transactionId: purchasedData?.transactionId,
        amount: priceAmount || 0,
        subscriptionFrequency:
          purchasedData?.productId === PRO_SUBSCRIPTION_ID
            ? SubscriptionFrequency.PRO
            : SubscriptionFrequency.PREMIUM,

        subscriptionStatus: SubscriptionStatus.COMPLETED,
        membershipAppliedAt,
      };

      await this.subscriptionLogRepo.updateOne(
        {
          transactionId: purchasedData?.transactionId,
        },
        logData,
        { upsert: true },
      );

      // update user's current subscription
      await this.userRepository.updateOne(
        { _id: user.userId },
        {
          currentSubscription: {
            ...subscriptionData,
            isCancelled: !!purchasedData?.cancellationDate,
          },
        },
      );

      return true;
    } catch (error: any) {
      throw error;
    }
  }

  /**
   * ${1:Description placeholder}
   *
   * @async
   * @param {*} user
   * @param {AndroidReceipt} bodyData
   * @returns {Promise<boolean>}
   */
  async verifyAndroidSubscription(user, @Args('bodyData') bodyData: AndroidReceipt) {
    const packageName = bodyData.packageName;
    const subscriptionId = bodyData.subscriptionId;
    const token = bodyData.token;
    // const userId = user._id;
    // const signature = bodyData?.signature;

    try {
      const data = await this.iap.validateAndroidReceipt({
        subscription: true,
        packageName,
        productId: subscriptionId,
        purchaseToken: token,
      });

      if (data instanceof Error) {
        throw new HttpException(
          data?.message || this.i18nService.t('subscription.receipt_error'),
          HttpStatus.UNPROCESSABLE_ENTITY,
        );
      }

      const purchaseData = data?.purchaseData[0];

      const transactionId = purchaseData?.purchaseToken || purchaseData?.transactionId;

      const subscriptionData = {
        purchasePlatform: PurchasePlatform.ANDROID,
        istrailPeriod: !!purchaseData?.trialPeriod,
        autoRenewStatus: !!purchaseData?.autoRenewing,
        originalTransactionId: transactionId,
        purchaseDate: purchaseData?.startTimeMillis ? new Date(purchaseData.startTimeMillis) : null,
        cancellationDate: purchaseData?.cancellationDate
          ? new Date(Number(purchaseData?.cancellationDate))
          : null,
        expirationDate: purchaseData?.expiryTimeMillis
          ? new Date(purchaseData?.expiryTimeMillis)
          : null,
      };

      const subscription = await this.subscriptionRepository.findOne({
        userId: user.userId.toString(),
        productId: purchaseData?.productId,
        originalTransactionId: transactionId,
      });

      if (subscription) {
        // update with latest info
        await this.subscriptionRepository.updateOne(
          {
            userId: user.userId.toString(),
            productId: purchaseData?.productId,
            originalTransactionId: transactionId,
          },
          {
            ...subscriptionData,
            productId: subscriptionId,
            receipt: token,
            transactionId: purchaseData?.orderId,
            packageName,
          },
        );
      } else {
        // if not found insert
        await this.subscriptionRepository.create({
          userId: user.userId.toString(),
          receipt: token,
          productId: subscriptionId,
          transactionId: purchaseData?.orderId,
          packageName,
          ...subscriptionData,
        });
      }

      let priceAmount; // in cents
      for (const sub of IAP_SUBSCRIPTION) {
        if (sub.productId === purchaseData?.productId) {
          priceAmount = sub.price.amount * 100;
        }
      }

      const membershipAppliedAt = purchaseData?.startTimeMillis
        ? new Date(purchaseData?.startTimeMillis)
        : new Date();

      const logData = {
        userId: user.userId.toString(),
        username: user.authProviderId || '',
        transactionId: purchaseData?.transactionId,
        amount: priceAmount || 0,
        subscriptionFrequency:
          purchaseData?.productId === PRO_SUBSCRIPTION_ID
            ? SubscriptionFrequency.PRO
            : SubscriptionFrequency.PREMIUM,

        subscriptionStatus: SubscriptionStatus.COMPLETED,
        membershipAppliedAt,
      };

      await this.subscriptionLogRepo.updateOne(
        {
          transactionId: purchaseData?.transactionId,
        },
        logData,
        { upsert: true },
      );
      // update user's current subscription
      await this.userRepository.updateOne(
        { _id: user.userId },
        {
          currentSubscription: {
            ...subscriptionData,
            isCancelled: !!purchaseData?.cancellationDate,
          },
        },
      );

      return !!data;
    } catch (error: any) {
      throw error;
    }
  }

  /**
   * ${1:Description placeholder}
   *
   * @async
   * @param {*} data
   * @returns {Promise<string>}
   */
  async handleIosWebhook(data: any) {
    try {
      const notification_type = data.notification_type;
      const auto_renew_status = data.auto_renew_status;

      const unifiedReceipt = data.unified_receipt;

      const validatedData = await this.iap.validateIosReceipt(unifiedReceipt.latest_receipt);

      const purchasedData = validatedData?.purchaseData[0];

      let createLog, priceAmount, payementStatus;
      if (notification_type === NOTIFICATION_TYPE.CANCEL || auto_renew_status == 'false') {
        priceAmount = 0;
        createLog = true;
        payementStatus = SubscriptionStatus.CANCELED;
        await this.subscriptionRepository.updateMany(
          {
            originalTransactionId: validatedData.purchaseData[0].originalTransactionId,
            productId: validatedData.purchaseData[0].productId,
          },
          {
            autoRenewStatus: false,
            cancellationDate: purchasedData?.cancellationDate
              ? new Date(purchasedData.cancellationDate)
              : new Date(),
            subscriptionStatus: false,
            notificationType: notification_type,
          },
        );
      }
      if (createLog) {
        const logData = {
          // userId: '', // need to determine
          username: '', // need to determine
          transactionId: purchasedData?.transactionId,
          amount: priceAmount,
          subscriptionFrequency:
            purchasedData?.productId === PRO_SUBSCRIPTION_ID
              ? SubscriptionFrequency.PRO
              : SubscriptionFrequency.PREMIUM,

          subscriptionStatus: payementStatus,
          membershipAppliedAt: purchasedData?.purchaseDateMs
            ? new Date(purchasedData.purchaseDateMs)
            : new Date(),
        };

        await this.subscriptionLogRepo.updateOne(
          {
            transactionId: purchasedData?.transactionId,
          },
          logData,
          { upsert: true },
        );
      }

      return 'OK';
    } catch (e) {
      throw new BadRequestException(e?.message);
    }
  }

  /**
   * ${1:Description placeholder}
   *
   * @async
   * @param {*} body
   * @returns {Promise<any>}
   */
  async handleAndroidWebhook(body: any) {
    try {
      const encodedData = body.message.data;
      const decodedData = Buffer.from(`${encodedData}`, 'base64');
      const msgRetrive = JSON.parse(decodedData.toString());

      const packageName = msgRetrive.packageName;
      const subscriptionId = msgRetrive.subscriptionNotification.subscriptionId;
      const token = msgRetrive.subscriptionNotification.purchaseToken;

      const data = await this.iap.validateAndroidReceipt({
        subscription: true,
        packageName,
        productId: subscriptionId,
        purchaseToken: token,
      });

      const purchaseData = data?.purchaseData[0];

      let createLog, priceAmount, payementStatus;
      if (msgRetrive.subscriptionNotification.notificationType == 3) {
        payementStatus = SubscriptionStatus.CANCELED;
        createLog = true;
        priceAmount = 0;
        await this.subscriptionRepository.updateMany(
          {
            originalTransactionId: data.purchaseData[0].orderId,
            productId: data.purchaseData[0].productId,
          },
          {
            ...data,
            cancellationDate: purchaseData?.cancellationDate
              ? new Date(purchaseData.cancellationDate)
              : new Date(),
            subscriptionStatus: false,
          },
        );
      }
      if (createLog) {
        const logData = {
          // userId: '',  need to determine
          username: '', // need to determine
          transactionId: purchaseData?.transactionId,
          amount: priceAmount,
          subscriptionFrequency:
            purchaseData?.productId === PRO_SUBSCRIPTION_ID
              ? SubscriptionFrequency.PRO
              : SubscriptionFrequency.PREMIUM,

          subscriptionStatus: payementStatus,
          membershipAppliedAt: purchaseData?.startTimeMillis
            ? new Date(purchaseData.startTimeMillis)
            : new Date(),
        };

        await this.subscriptionLogRepo.updateOne(
          {
            transactionId: purchaseData?.transactionId,
          },
          logData,
          { upsert: true },
        );
      }
      return 'OK';
    } catch (error) {
      return error?.message;
    }
  }
